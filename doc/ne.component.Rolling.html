롤링 데이터값
Example
var roll = new ne.component.Rolling({
     element: document.getElementById('rolling'),
     initNum: 0,
     direction: 'horizontal',
     isVariable: true,
     unit: 'page',
     isAuto: false,
     motion: 'easeInOut',
     duration:2000
}, ['<div>data1</div>','<div>data2</div>', '<div>data3</div>']);
