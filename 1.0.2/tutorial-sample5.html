        // HTML 데이터
    var htmlData = '<div class="panel-heading pan">default Content</div>',
            nextData = '<div class="panel-heading pan">nextData</div>',
            prevData = '<div class="panel-heading pan">prevData</div>';

    // 생성
    var rolling1 = new tui.component.Rolling({
        element: document.getElementById('rolling'),
        initNum: 0,
        direction: 'horizontal',
        isVariable: true,
        duration: 400,
        motion: 'linear',
        isCircle: false
    }, htmlData);

    // 이벤트 등록
    var control = document.getElementById('control');
    control.onclick = function(e) {
        var e = e || window.event,
                target = e.target || e.srcElement,
                value;
        if (target.tagName.toLowerCase() !== 'button') {
            return;
        }
        className = target.className;
        if (className.indexOf('left') > -1) {
            rolling1.roll(nextData, 'prev');
        } else {
            rolling1.roll(prevData, 'next');
        }
    };

    // 커스텀 이벤트 등록
    var statusElement = document.getElementById('status');
    rolling1.attach('beforeMove', function() {
        statusElement.style.backgroundColor = 'yellow';
        statusElement.innerHTML = '상태 : 무브';
    });
    rolling1.attach('afterMove', function() {
        statusElement.style.backgroundColor = 'green';
        statusElement.innerHTML = '상태 : 무브끝';
    });
